This functions counts how many values from a list must be summed in order to reach some limit: 

def count_to_limit(values, limit):
    current = 0
    number = 0
    for v in values:
        current += v
        number += 1
        if current >= limit:
            return number

print count_to_limit([1, 2, 3, 5, 7], 8)
